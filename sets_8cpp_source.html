19bool Interval::is_canonical(const RCP<const Number> &s,
20 const RCP<const Number> &e, bool left_open,
21 bool right_open)
22{
23 if (is_a<Complex>(*s) or is_a<Complex>(*e))
24 throw NotImplementedError("Complex set not implemented");
25 if (eq(*e, *s)) {
26 return false;
27 } else if (eq(*min({s, e}), *e)) {
28 return false;
29 }
30 return true;

54int Interval::compare(const Basic &s) const
55{
56 // compares two interval based on their length
57 SYMENGINE_ASSERT(is_a<Interval>(s))
58 const Interval &o = down_cast<const Interval &>(s);
59 if (left_open_ and not o.left_open_) {
60 return -1;
61 } else if (not left_open_ and o.left_open_) {
62 return 1;
63 } else if (right_open_ and not o.right_open_) {
64 return 1;
65 } else if (not right_open_ and o.right_open_) {
66 return -1;
67 } else {
68 auto temp = start_->__cmp__(*(o.start_));
69 if (temp != 0) {
70 return temp;
71 } else {
72 return end_->__cmp__(*(o.end_));
73 }
74 }

115RCP<const Set> Interval::set_intersection(const RCP<const Set> &o) const
116{
117 if (is_a<Interval>(*o)) {
118 const Interval &other = down_cast<const Interval &>(*o);
119 RCP<const Number> start, end;
120 bool left_open, right_open;
121 RCP<const Basic> start_end, end_start;
122 start_end = min({this->start_, other.end_});
123 end_start = min({this->end_, other.start_});
124
125 if (eq(*this->start_, *start_end) and eq(*other.start_, *end_start)) {
126 RCP<const Basic> start_start, end_end;
127 start_start = min({this->start_, other.start_});
128 end_end = min({this->end_, other.end_});
129 if (neq(*this->start_, *other.start_)) {
130 if (eq(*this->start_, *start_start)) {
131 start = other.start_;
132 left_open = other.left_open_;
133 } else {
134 start = this->start_;
135 left_open = this->left_open_;
136 }
137 } else {
138 start = this->start_;
139 left_open = this->left_open_ or other.left_open_;
140 }
141
142 if (neq(*this->end_, *other.end_)) {
143 if (eq(*this->end_, *end_end)) {
144 end = this->end_;
145 right_open = this->right_open_;
146 } else {
147 end = other.end_;
148 right_open = other.right_open_;
149 }
150 } else {
151 end = this->end_;
152 right_open = this->right_open_ or other.right_open_;
153 }
154 return interval(start, end, left_open, right_open);
155 } else {
156 return emptyset();
157 }
158 }
159 if (is_a<Integers>(*o)) {
160 if (is_a_Number(*start_) and is_a_Number(*end_)) {
161 auto first = SymEngine::ceiling(start_);
162 auto last = SymEngine::floor(end_);
163 if (eq(*first, *start_) and left_open_) {
164 first = add(first, integer(1));
165 }
166 if (eq(*last, *end_) and right_open_) {
167 last = add(last, integer(-1));
168 }
169 if (eq(*Lt(last, first), *boolTrue)) {
170 return emptyset();
171 }
172 set_basic container;
173 while (eq(*Ge(last, first), *boolTrue)) {
174 container.insert(first);
175 first = add(first, integer(1));
176 }
177 return finiteset(container);
178 } else {
179 return SymEngine::set_intersection(
180 {rcp_from_this_cast<const Set>(), o});
181 }
182 }
183 if (is_a<UniversalSet>(*o) or is_a<EmptySet>(*o) or is_a<FiniteSet>(*o)
184 or is_a<Union>(*o) or is_a<Rationals>(*o) or is_a<Reals>(*o)
185 or is_a<Complexes>(*o)) {
186 return (*o).set_intersection(rcp_from_this_cast<const Set>());
187 }
188 throw SymEngineException("Not implemented Intersection class");
189}

199RCP<const Set> Interval::set_union(const RCP<const Set> &o) const
200{
201 if (is_a<Interval>(*o)) {
202 const Interval &other = down_cast<const Interval &>(*o);
203 RCP<const Basic> start_start, end_end, m;
204 RCP<const Number> start, end;
205 bool left_open, right_open;
206 start_start = max({this->start_, other.start_});
207 end_end = min({this->end_, other.end_});
208 m = min({start_start, end_end});
209 if ((eq(*end_end, *start_start) and eq(*end_end, *m)
210 and ((eq(*end_end, *this->end_) and this->right_open_)
211 or (eq(*end_end, *other.end_) and other.right_open_)))
212 or (eq(*end_end, *m) and not eq(*end_end, *start_start))) {
213 return SymEngine::make_set_union(
214 {rcp_from_this_cast<const Set>(), o});
215 } else {
216 if (eq(*min({this->start_, other.start_}), *this->start_)) {
217 start = this->start_;
218 } else {
219 start = other.start_;
220 }
221 if (eq(*max({this->end_, other.end_}), *this->end_)) {
222 end = this->end_;
223 } else {
224 end = other.end_;
225 }
226 left_open = ((neq(*this->start_, *start) or this->left_open_)
227 and (neq(*other.start_, *start) or other.left_open_));
228 right_open = ((neq(*this->end_, *end) or this->right_open_)
229 and (neq(*other.end_, *end) or other.right_open_));
230 return interval(start, end, left_open, right_open);
231 }
232 }
233 if (is_a<UniversalSet>(*o) or is_a<EmptySet>(*o) or is_a<FiniteSet>(*o)
234 or is_a<Union>(*o) or is_a<Complexes>(*o) or is_a<Reals>(*o)
235 or is_a<Rationals>(*o) or is_a<Integers>(*o)) {
236 return (*o).set_union(rcp_from_this_cast<const Set>());
237 }
238 return SymEngine::make_set_union({rcp_from_this_cast<const Set>(), o});
239}

241RCP<const Set> Interval::set_complement(const RCP<const Set> &o) const
242{
243 if (is_a<Interval>(*o)) {
244 set_set cont;
245 const Interval &other = down_cast<const Interval &>(*o);
246 if (eq(*max({start_, other.start_}), *start_)) {
247 cont.insert(interval(other.get_start(), start_,
248 other.get_left_open(), not left_open_));
249 }
250 if (eq(*min({end_, other.end_}), *end_)) {
251 cont.insert(interval(end_, other.get_end(), not right_open_,
252 other.get_right_open()));
253 }
254 return SymEngine::set_union(cont);
255 }
256 return SymEngine::set_complement_helper(rcp_from_this_cast<const Set>(), o);
257}

689RCP<const Boolean> FiniteSet::contains(const RCP<const Basic> &a) const
690{
691 set_basic rest;
692 for (const auto &elem : container_) {
693 auto cont = Eq(elem, a);
694 if (eq(*cont, *boolTrue))
695 return boolTrue;
696 if (not eq(*cont, *boolFalse))
697 rest.insert(elem);
698 }
699 if (rest.empty()) {
700 return boolFalse;
701 } else {
702 return make_rcp<Contains>(a, finiteset(rest));
703 }
704}
705
706RCP<const Set> FiniteSet::set_union(const RCP<const Set> &o) const
707{
708 if (is_a<FiniteSet>(*o)) {
709 const FiniteSet &other = down_cast<const FiniteSet &>(*o);
710 set_basic container;
711 std::set_union(container_.begin(), container_.end(),
712 other.container_.begin(), other.container_.end(),
713 std::inserter(container, container.begin()),
714 RCPBasicKeyLess{});
715 return finiteset(container);
716 }
717 if (is_a<Interval>(*o)) {
718 set_basic container;
719 const Interval &other = down_cast<const Interval &>(*o);
720 bool left = other.get_left_open(), right = other.get_right_open();
721 for (const auto &a : container_) {
722 auto contain = o->contains(a);
723 if (eq(*contain, *boolFalse)) {
724 if (left)
725 if (eq(*other.get_start(), *a)) {
726 left = false;
727 continue;
728 }
729 if (right)
730 if (eq(*other.get_end(), *a)) {
731 right = false;
732 continue;
733 }
734 container.insert(a);
735 } else if (is_a<Contains>(*contain)) {
736 container.insert(a);
737 }
738 }
739 if (not container.empty()) {
740 if (left == other.get_left_open()
741 and right == other.get_right_open()) {
742 return SymEngine::make_set_union({finiteset(container), o});
743 } else {
744 return SymEngine::make_set_union(
745 set_set({finiteset(container),
746 interval(other.get_start(), other.get_end(), left,
747 right)}));
748 }
749 } else {
750 if (left == other.get_left_open()
751 and right == other.get_right_open()) {
752 return o;
753 } else {
754 return interval(other.get_start(), other.get_end(), left,
755 right);
756 }
757 }
758 }
759 if (is_a<Complexes>(*o)) {
760 set_basic container;
761 for (const auto &elem : container_) {
762 if (!is_a_Number(*elem)) {
763 container.insert(elem);
764 }
765 }
766 if (container.empty()) {
767 return complexes();
768 } else {
769 return SymEngine::make_set_union(
770 {complexes(), finiteset(container)});
771 }
772 }
773 if (is_a<Reals>(*o)) {
774 set_basic container;
775 for (const auto &elem : container_) {
776 if (!is_a_Number(*elem)
777 || down_cast<const Number &>(*elem).is_complex()) {
778 container.insert(elem);
779 }
780 }
781 if (container.empty()) {
782 return reals();
783 } else {
784 return SymEngine::make_set_union({reals(), finiteset(container)});
785 }
786 }
787 if (is_a<Rationals>(*o)) {
788 set_basic container;
789 for (const auto &elem : container_) {
790 if (!is_a_Number(*elem)
791 || down_cast<const Number &>(*elem).is_complex()) {
792 container.insert(elem);
793 }
794 }
795 if (container.empty()) {
796 return rationals();
797 } else {
798 return SymEngine::make_set_union(
799 {rationals(), finiteset(container)});
800 }
801 }
802 if (is_a<Integers>(*o)) {
803 set_basic container;
804 for (const auto &elem : container_) {
805 if (!is_a<Integer>(*elem)) {
806 container.insert(elem);
807 }
808 }
809 if (container.empty()) {
810 return integers();
811 } else {
812 return SymEngine::make_set_union(
813 {integers(), finiteset(container)});
814 }
815 }
816 if (is_a<UniversalSet>(*o) or is_a<EmptySet>(*o) or is_a<Union>(*o)) {
817 return (*o).set_union(rcp_from_this_cast<const Set>());
818 }
819 return SymEngine::make_set_union({rcp_from_this_cast<const Set>(), o});
820}

822RCP<const Set> FiniteSet::set_intersection(const RCP<const Set> &o) const
823{
824 if (is_a<FiniteSet>(*o)) {
825 const FiniteSet &other = down_cast<const FiniteSet &>(*o);
826 set_basic container;
827 std::set_intersection(container_.begin(), container_.end(),
828 other.container_.begin(), other.container_.end(),
829 std::inserter(container, container.begin()),
830 RCPBasicKeyLess{});
831 return finiteset(container);
832 }
833 if (is_a<Interval>(*o)) {
834 set_basic container;
835 for (const auto &a : container_) {
836 auto contain = o->contains(a);
837 if (eq(*contain, *boolTrue))
838 container.insert(a);
839 if (is_a<Contains>(*contain))
840 throw SymEngineException("Not implemented");
841 }
842 return finiteset(container);
843 }
844 if (is_a<Complexes>(*o) or is_a<Reals>(*o) or is_a<Rationals>(*o)) {
845 set_basic kept;
846 set_basic others;
847 for (const auto &elem : container_) {
848 if (is_a_Number(*elem)) {
849 if (!down_cast<const Number &>(*elem).is_complex()) {
850 if (!is_a<Rationals>(*o)
851 or down_cast<const Number &>(*elem).is_exact()) {
852 kept.insert(elem);
853 }
854 } else {
855 if (is_a<Complexes>(*o)) {
856 kept.insert(elem);
857 }
858 }
859 } else {
860 others.insert(elem);
861 }
862 }
863 if (kept.empty()) {
864 if (others.empty()) {
865 return emptyset();
866 } else {
867 return SymEngine::set_intersection({o, finiteset(others)});
868 }
869 } else {
870 if (others.empty()) {
871 return finiteset(kept);
872 } else {
873 others.insert(kept.begin(), kept.end());
874 return SymEngine::set_intersection({o, finiteset(others)});
875 }
876 }
877 }
878 if (is_a<Integers>(*o)) {
879 set_basic kept_integers;
880 set_basic others;
881 for (const auto &elem : container_) {
882 if (is_a_Number(*elem)) {
883 if (is_a<Integer>(*elem)) {
884 kept_integers.insert(elem);
885 }
886 } else {
887 others.insert(elem);
888 }
889 }
890 if (kept_integers.empty()) {
891 if (others.empty()) {
892 return emptyset();
893 } else {
894 return SymEngine::set_intersection(
895 {integers(), finiteset(others)});
896 }
897 } else {
898 if (others.empty()) {
899 return finiteset(kept_integers);
900 } else {
901 others.insert(kept_integers.begin(), kept_integers.end());
902 return SymEngine::set_intersection(
903 {integers(), finiteset(others)});
904 }
905 }
906 }
907 if (is_a<UniversalSet>(*o) or is_a<EmptySet>(*o) or is_a<Union>(*o)) {
908 return (*o).set_intersection(rcp_from_this_cast<const Set>());
909 }
910 throw SymEngineException("Not implemented Intersection class");
911}

913RCP<const Set> FiniteSet::set_complement(const RCP<const Set> &o) const
914{
915 if (is_a<FiniteSet>(*o)) {
916 const FiniteSet &other = down_cast<const FiniteSet &>(*o);
917 set_basic container;
918 std::set_difference(other.container_.begin(), other.container_.end(),
919 container_.begin(), container_.end(),
920 std::inserter(container, container.begin()),
921 RCPBasicKeyLess{});
922 return finiteset(container);
923 }
924
925 if (is_a<Interval>(*o)) {
926 set_set intervals;
927 auto &other = down_cast<const Interval &>(*o);
928 RCP<const Number> last = other.get_start();
929 RCP<const Number> a_num;
930 set_basic rest;
931 bool left_open = other.get_left_open(),
932 right_open = other.get_right_open();
933 for (auto it = container_.begin(); it != container_.end(); it++) {
934 if (eq(*max({*it, other.get_start()}), *other.get_start())) {
935 if (eq(**it, *other.get_start()))
936 left_open = true;
937 continue;
938 }
939 if (eq(*max({*it, other.get_end()}), **it)) {
940 if (eq(**it, *other.get_end()))
941 right_open = true;
942 break;
943 }
944 if (is_a_Number(**it)) {
945 a_num = rcp_static_cast<const Number>(*it);
946 intervals.insert(interval(last, a_num, left_open, true));
947 last = a_num;
948 left_open = true;
949 } else {
950 rest.insert(*it);
951 }
952 }
953
954 if (eq(*max({last, other.get_end()}), *other.get_end())) {
955 intervals.insert(
956 interval(last, other.get_end(), left_open, right_open));
957 }
958 if (rest.empty()) {
959 return SymEngine::make_set_union(intervals);
960 } else {
961 return make_rcp<const Complement>(
962 SymEngine::make_set_union(intervals), finiteset(rest));
963 }
964 }
965
966 return SymEngine::set_complement_helper(rcp_from_this_cast<const Set>(), o);
967}

1296
1297RCP<const Set> set_union(const set_set &in)
1298{
1299 set_set input;
1300 set_basic combined_FiniteSet;
1301 for (auto it = in.begin(); it != in.end(); ++it) {
1302 if (is_a<FiniteSet>(**it)) {
1303 const FiniteSet &other = down_cast<const FiniteSet &>(**it);
1304 combined_FiniteSet.insert(other.get_container().begin(),
1305 other.get_container().end());
1306 } else if (is_a<UniversalSet>(**it)) {
1307 return universalset();
1308 } else if (not is_a<EmptySet>(**it)) {
1309 input.insert(*it);
1310 }
1311 }
1312 if (input.empty()) {
1313 return finiteset(combined_FiniteSet);
1314 } else if (input.size() == 1 && combined_FiniteSet.empty()) {
1315 return *input.begin();
1316 }
1317 // Now we rely on respective containers' own rules
1318 // TODO: Improve it to O(log n)
1319 RCP<const Set> combined_Rest = finiteset(combined_FiniteSet);
1320 for (auto it = input.begin(); it != input.end(); ++it) {
1321 combined_Rest = combined_Rest->set_union(*it);
1322 }
1323 return combined_Rest;
1324}
1325
1326RCP<const Set> set_intersection(const set_set &in)
1327{
1328 // https://en.wikipedia.org/wiki/Intersection_(set_theory)#Nullary_intersection
1329 if (in.empty())
1330 return universalset();
1331
1332 // Global rules
1333 // If found any emptyset then return emptyset
1334 set_set incopy;
1335 for (const auto &input : in) {
1336 if (is_a<EmptySet>(*input)) {
1337 return emptyset();
1338 } else if (not is_a<UniversalSet>(*input)) {
1339 incopy.insert(input);
1340 }
1341 }
1342
1343 if (incopy.empty())
1344 return universalset();
1345 if (incopy.size() == 1)
1346 return *incopy.begin();
1347
1348 // Handle finite sets
1349 std::vector<RCP<const Set>> fsets, othersets;
1350 for (const auto &input : incopy) {
1351 if (is_a<FiniteSet>(*input)) {
1352 fsets.push_back(input);
1353 } else {
1354 othersets.push_back(input);
1355 }
1356 }
1357 if (fsets.size() != 0) {
1358 const FiniteSet &fs = down_cast<const FiniteSet &>(**fsets.begin());
1359 auto cont = fs.get_container();
1360 fsets.erase(fsets.begin());
1361 set_basic finalfs;
1362 for (const auto &fselement : cont) {
1363 bool present = true;
1364 for (const auto &fset : fsets) {
1365 auto contain = fset->contains(fselement);
1366 if (not(eq(*contain, *boolTrue) or eq(*contain, *boolFalse))) {
1367 throw SymEngineException(
1368 "Not implemented Intersection class");
1369 }
1370 present = present and eq(*contain, *boolTrue);
1371 }
1372 if (!present)
1373 continue;
1374 for (const auto &oset : othersets) {
1375 auto contain = oset->contains(fselement);
1376 if (not(eq(*contain, *boolTrue) or eq(*contain, *boolFalse))) {
1377 throw SymEngineException(
1378 "Not implemented Intersection class");
1379 }
1380 present = present and eq(*contain, *boolTrue);
1381 }
1382 if (present)
1383 finalfs.insert(fselement);
1384 }
1385 return finiteset(finalfs);
1386 }
1387
1388 // If any of the sets is union, then return a Union of Intersections
1389 for (auto it = incopy.begin(); it != incopy.end(); ++it) {
1390 if (is_a<Union>(**it)) {
1391 auto container = down_cast<const Union &>(**it).get_container();
1392 incopy.erase(it);
1393 auto other = SymEngine::set_intersection(incopy);
1394 set_set usets;
1395 for (const auto &c : container) {
1396 usets.insert(SymEngine::set_intersection({c, other}));
1397 }
1398 return SymEngine::set_union(usets);
1399 }
1400 }
1401
1402 // Simplify and return a `Complement` if any of the sets is a complement
1403 for (auto it = incopy.begin(); it != incopy.end(); ++it) {
1404 if (is_a<Complement>(**it)) {
1405 auto container
1406 = down_cast<const Complement &>(**it).get_container();
1407 auto universe = down_cast<const Complement &>(**it).get_universe();
1408 incopy.erase(it);
1409 incopy.insert(universe);
1410 auto other = SymEngine::set_intersection(incopy);
1411 return SymEngine::set_complement(other, container);
1412 }
1413 }
1414
1415 // Pair-wise rules
1416 // TO-DO: needs the following improvement once Intersection
1417 // class is implemented.
1418 // input_oset if found to be not simplified, then skip this
1419 // pair.
1420 if (incopy.size() > 1) {
1421 auto temp = *incopy.begin();
1422 auto it = std::next(incopy.begin());
1423 for (; it != incopy.end(); ++it) {
1424 temp = temp->set_intersection(*it);
1425 }
1426 return temp;
1427 }
1428
1429 if (incopy.size() == 1) {
1430 return *incopy.begin();
1431 } else {
1432 throw SymEngineException("Not implemented Intersection class");
1433 }
1434}

1437RCP<const Set> set_complement_helper(const RCP<const Set> &container,
1438 const RCP<const Set> &universe)
1439{
1440 if (is_a<Union>(*universe)) {
1441 auto univ = down_cast<const Union &>(*universe).get_container();
1442 set_set container_;
1443 for (auto &a : univ) {
1444 container_.insert(container->set_complement(a));
1445 }
1446 return SymEngine::set_union(container_);
1447 } else if (is_a<EmptySet>(*universe)) {
1448 return emptyset();
1449 } else if (is_a<FiniteSet>(*universe)) {
1450 const FiniteSet &other = down_cast<const FiniteSet &>(*universe);
1451 set_basic container_;
1452 set_basic rest;
1453 for (const auto &a : other.get_container()) {
1454 auto contain = container->contains(a);
1455 if (eq(*contain, *boolFalse)) {
1456 container_.insert(a);
1457 } else if (is_a<Contains>(*contain)) {
1458 rest.insert(a);
1459 }
1460 }
1461 if (rest.empty()) {
1462 return finiteset(container_);
1463 } else {
1464 return SymEngine::set_union(
1465 {finiteset(container_),
1466 make_rcp<const Complement>(finiteset(rest), container)});
1467 }
1468 }
1469 return make_rcp<const Complement>(universe, container);
1470}

1479RCP<const Set> conditionset(const RCP<const Basic> &sym,
1480 const RCP<const Boolean> &condition)
1481{
1482 if (eq(*condition, *boolean(false))) {
1483 return emptyset();
1484 } else if (eq(*condition, *boolean(true))) {
1485 return universalset();
1486 }
1487 if (is_a<And>(*condition)) {
1488 auto cont = down_cast<const And &>(*condition).get_container();
1489 set_boolean newcont;
1490 set_basic present, others;
1491 for (auto it = cont.begin(); it != cont.end(); it++) {
1492 if (is_a<Contains>(**it)
1493 and eq(*down_cast<const Contains &>(**it).get_expr(), *sym)
1494 and is_a<FiniteSet>(
1495 *down_cast<const Contains &>(**it).get_set())) {
1496 auto fset = down_cast<const Contains &>(**it).get_set();
1497 auto fcont
1498 = down_cast<const FiniteSet &>(*fset).get_container();
1499 // use the result of simplification done in `logical_and()`
1500 for (const auto &elem : fcont) {
1501 if (not(is_a_Number(*elem) or is_a<Constant>(*elem))) {
1502 others.insert(elem);
1503 } else {
1504 // logical_and() doesn't guarantee that if element of a
1505 // finiteset is a number, then it satisfies other
1506 // conditions
1507 // it only assures that there doesn't exist any such
1508 // element of finiteset that surely fails in other
1509 // conditions.
1510 auto restCont = cont;
1511 restCont.erase(*it);
1512 auto restCond = logical_and(restCont);
1513 map_basic_basic d;
1514 d[sym] = elem;
1515 auto contain = restCond->subs(d);
1516 if (eq(*contain, *boolean(true))) {
1517 present.insert(elem);
1518 } else if (not eq(*contain, *boolean(false))) {
1519 others.insert(elem);
1520 } else {
1521 throw SymEngineException("element should have "
1522 "been removed within "
1523 "logical_and()");
1524 }
1525 }
1526 }
1527 } else {
1528 newcont.insert(*it);
1529 }
1530 }
1531 if (not present.empty()) {
1532 newcont.insert(finiteset(others)->contains(sym));
1533 return SymEngine::set_union(
1534 {finiteset(present), conditionset(sym, logical_and(newcont))});
1535 }
1536 }
1537 if (is_a<Contains>(*condition)) {
1538 return down_cast<const Contains &>(*condition).get_set();
1539 }
1540 return make_rcp<const ConditionSet>(sym, condition);
1541}
